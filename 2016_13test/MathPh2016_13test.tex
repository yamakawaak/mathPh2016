\begin{enumerate}

\item $\displaystyle{\frac{\partial u }{\partial x} = x+y}$
を満たし，$x=0$で$u=e^{-y}$に従って指数関数的に減少する解を求めよ．\\
\\
（解答例）\\
\begin{eqnarray*}
&&\frac{\partial u }{\partial x} = x+y \\[5pt]
&\Leftrightarrow& \ u = \frac{1}{2}x^2 + xy + f(y).
\end{eqnarray*}
$x=0$における条件式を代入すると，
\begin{eqnarray*}
&& e^{-y} = f(y). \\[5pt]
&\therefore& u = \frac{1}{2}x^2+ xy + e^{-y}.
\end{eqnarray*}

\end{enumerate}
